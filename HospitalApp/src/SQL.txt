CREATE TABLE PERSON(
	ID NUMBER PRIMARY KEY,
	NAME NVARCHAR2(30),
	FAMILY NVARCHAR2(30),
	NATIONAL_CODE CHAR(10),
	BIRTH_DATE DATE,
	FATHER NVARCHAR2(30),
 	TEL CHAR(11),
	ADDRESS NVARCHAR2(100),
 	ROLE_ID REFERENCES ROLE,
	PHOTO VARCHAR2(50),
	DESCRIPTION NVARCHAR2(100),
	GENDER NUMBER(1));

CREATE SEQUENCE PERSON_SEQ START WITH 1 INCREMENT  BY 1;




CREATE TABLE ROLE (
	ID NUMBER PRIMARY KEY,
	NAME NVARCHAR2(30),
	DESCRIPTION NVARCHAR2(100));

CREATE SEQUENCE ROLE_SEQ START WITH 1 INCREMENT  BY 1;




CREATE TABLE MEDICINE(
	ID NUMBER PRIMARY KEY,
	NAME NVARCHAR2(30),
	DESCRIPTION NVARCHAR2(100)); 

CREATE SEQUENCE MEDICINE_SEQ START WITH 1 INCREMENT  BY 1;




CREATE TABLE MEDICAL_RECORDS(
	ID NUMBER PRIMARY KEY,
	TYPE_DISEASE NVARCHAR2(50),
	DURATION_TREATMENT NVARCHAR2(50),
	DOCTORS_NAME NVARCHAR2(50),
	MEDICINE_ID REFERENCES MEDICINE);

CREATE SEQUENCE MEDICAL_RECORDS_SEQ START WITH 1 INCREMENT  BY 1;




CREATE TABLE PATIENT(
	PERSON_ID REFERENCES PERSON,
	BLOOD_TYPE NUMBER(1),
	HEIGHT NUMBER(3,3),
	WEIGHT NUMBER(3,3),
	MEDICAL_RECORDS_ID REFERENCES MEDICAL_RECORDS);

CREATE SEQUENCE PATIENT_SEQ START WITH 1 INCREMENT  BY 1;




CREATE TABLE EQUIPMENT(
	ID NUMBER PRIMARY KEY,
	NAME NVARCHAR2(30),
	DESCRIPTION NVARCHAR2(100));

CREATE SEQUENCE EQUIPMENT_CODE_SEQ START WITH 1 INCREMENT  BY 1;




CREATE TABLE ROOM(
	ID NUMBER PRIMARY KEY,
	FLOOR NUMBER(2),
	ROOM_CODE NUMBER(2)
	EQUIPMENT_CODE_ID REFERENCES EQUIPMENT_CODE);

CREATE SEQUENCE ROOM_SEQ START WITH 1 INCREMENT  BY 1;



CREATE TABLE DOCTOR(
	PERSON_ID REFERENCES PERSON,
	MEDICAL_SYSTEM_CODE CHAR(),

CREATE SEQUENCE DOCTOR_SEQ START WITH 1 INCREMENT  BY 1;



CREATE TABLE USER(
	ID NUMBER PRIMARY KEY,
	PERSON_ID REFERENCES PERSON,
	USER_NAME NVARCHAR2(30),
	PASSWORD NVARCHAR2(30),
	ACCESS_KEY VARCHAR2(20));  

CREATE SEQUENCE USER_SEQ START WITH 1 INCREMENT  BY 1;





CREATE TABLE VISIT(
	PATIENT_ID REFERENCES PATIENT,
	ROOM_ID REFERENCES ROOM,
	DOCTOR_ID REFERENCES DOCTOR,	
	DATE_TIME TIMESTAMP,
	PAYMENT NUMBER ,
	TURN NUMBER(1),
	USER_ID REFERENCES USER);
	
CREATE SEQUENCE VISIT_SEQ START WITH 1 INCREMENT  BY 1;




CREATE TABLE EDUCATIONAL_RECORDS(
	ID NUMBER PRIMARY KEY,
	DOCTOR_ID REFERENCES DOCTOR,
	TITEL_DOCUMENT NVARCHAR2(30),
	UNIVERSITY NVARCHAR2(50),
	GPA NUMBER(2,2));
	
CREATE SEQUENCE EDUCATIONAL_RECORDS_SEQ START WITH 1 INCREMENT  BY 1;



CREATE TABLE RESUME(
	ID NUMBER PRIMARY KEY,
	JOB_TITLE NVARCHAR2(30),
	DOCTOR_ID REFERENCES DOCTOR,
	WORK_PLACE NVARCHAR2(30),
	START_DATE TIMESTAMP,
	END_DATE TIMESTAMP);

CREATE SEQUENCE RESUME_SEQ START WITH 1 INCREMENT  BY 1;



CREATE TABLE TIMEING(
	ID NUMBER PRIMARY KEY,
	DOCTOR_ID REFERENCES DOCTOR,
	ROOM_ID REFERENCES ROOM,
	START_DATE TIMESTAMP,
	END_DATE TIMESTAMP,
	DATE TIMESTAMP,
	DAY NVARCHAR2(10));

CREATE SEQUENCE TIMEING_SEQ START WITH 1 INCREMENT  BY 1;
